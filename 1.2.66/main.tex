%% ── EXACT SOLUTION ──────────────────────────────────────────────
\begin{frame}{\textbf{Exact Solution of the ODE}}
  \begin{center}
    \vfill
    The ODE $\dfrac{dy}{dx} = 2xy$ is a \textbf{separable first-order ODE}.

    \vspace{1em}
    Separate variables and integrate:
    \[
      \int \frac{dy}{y} \;=\; \int 2x\,dx
      \quad\Longrightarrow\quad
      \ln|y| \;=\; x^2 + C
    \]

    \vspace{0.8em}
    Applying the initial condition $y(0) = 1$:
    \[
      \ln(1) = 0 + C \;\Longrightarrow\; C = 0
    \]

    \vspace{1em}
    \Large\textbf{\textcolor{madblue}{$y(x) \;=\; e^{x^2}$}}

    \vspace{1em}
    \normalsize
    \textbf{Verification:}\quad
    $y(0.8) = e^{0.64} \approx \mathbf{1.8965}$\;

%% ── WHY ADAMS-BASHFORTH ─────────────────────────────────────────
\begin{frame}{\textbf{Why Adams--Bashforth? (Answer to Question 27)}}
  \begin{center}
    \vfill
    We have \textbf{four equally-spaced data points} $(x_0, x_1, x_2, x_3)$\\[0.3em]
    with step size $h = 0.2$.

    \vspace{1.5em}
    The \textbf{Adams--Bashforth family} is the standard class of\\
    \textbf{explicit linear multistep predictor methods} for IVPs.

    \vspace{1.5em}
    With 4 data points available, we can use the\\
    \textbf{highest-order} Adams--Bashforth formula:

    \vspace{1em}
    \Large\textbf{\textcolor{madblue}{4-Step Adams--Bashforth Predictor}}

    \vspace{1em}
    \normalsize
    This is an \textbf{explicit method} $\Rightarrow$ it is a \textit{predictor},\\[0.3em]
    because it computes $y_{n+1}$ using \textbf{only previous values}.

    \vspace{0.5em}
    \textbf{Key Property:} The method does not require solving implicit equations,\\[0.2em]
    making it computationally efficient for prediction steps.
    \vfill
  \end{center}
\end{frame}

%% ── AB4 DERIVATION: CONCEPT ────────────────────────────────────
\begin{frame}{\textbf{Derivation: Adams--Bashforth Concept}}
  \begin{center}
    \vfill
    Starting from the fundamental theorem of calculus:
    \[
      y(x_{n+1}) - y(x_n) \;=\; \int_{x_n}^{x_{n+1}} f\bigl(x, y(x)\bigr)\,dx
    \]

    \vspace{1em}
    \textbf{Key Idea:}\; Approximate $f(x, y)$ over $[x_n, x_{n+1}]$\\[0.3em]
    using a \textbf{backward Newton interpolating polynomial}\\[0.3em]
    through the past data points $f_n, f_{n-1}, f_{n-2}, \ldots$

    \vspace{1.5em}
    For the \textbf{4-step} variant, we fit a \textbf{cubic polynomial}\\[0.3em]
    through $(f_{n-3}, f_{n-2}, f_{n-1}, f_n)$ and integrate.

    \vspace{1em}
    \textbf{Method Characteristics:}
    \begin{itemize}
      \item \textit{Local truncation error: $\mathcal{O}(h^5)$}
      \item \textit{Global accuracy: 4th-order (error $\mathcal{O}(h^4)$)}
      \item \textit{Explicit formula: no iterations needed}
    \end{itemize}
    \vfill
  \end{center}
\end{frame}

%% ── AB4 FORMULA ────────────────────────────────────────────────
\begin{frame}{\textbf{Adams--Bashforth 4-Step Formula}}
  \begin{center}
    \vfill
    The \textbf{Adams--Bashforth 4-step predictor} formula is:
    \[
      \boxed{y_{n+1} \;=\; y_n \;+\; \frac{h}{24}\bigl[55f_n - 59f_{n-1} + 37f_{n-2} - 9f_{n-3}\bigr]}
    \]

    \vspace{1em}
    where $f_k = f(x_k, y_k) = 2 x_k y_k$.

    \vspace{1.5em}
    \textbf{Properties:}
    \begin{itemize}
      \item \textbf{Explicit:}\; computes $y_{n+1}$ directly (no iteration needed)
      \item \textbf{Order:}\; $4$ (local truncation error $= \mathcal{O}(h^5)$)
      \item \textbf{Starting values:}\; requires 4 prior values $(y_{n-3}, \ldots, y_n)$
      \item \textbf{Coefficients:}\; derived from integrating the Newton backward polynomial
    \end{itemize}
    \vfill
  \end{center}
\end{frame}

%% ── COMPUTING f VALUES ──────────────────────────────────────────
\begin{frame}{\textbf{Step 1: Compute Derivative Values $f_i$}}
  \begin{center}

    Using $f(x, y) = 2xy$, we compute the derivative at each data point:

    \begin{tabular}{cccl}

      $1$ & $0.2$ & $1.0408$ & $f_1 = 2(0.2)(1.0408) = \mathbf{0.41632}$ \\[0.3em]
      $2$ & $0.4$ & $1.1735$ & $f_2 = 2(0.4)(1.1735) = \mathbf{0.93880}$ \\[0.3em]
      $3$ & $0.6$ & $1.4333$ & $f_3 = 2(0.6)(1.4333) = \mathbf{1.71996}$ \\

  \end{center}
\end{frame}

%% ── SUBSTITUTION ────────────────────────────────────────────────
\begin{frame}{\textbf{Step 2: Apply the AB4 Formula}}

      y_4 \;=\; y_3 + \frac{h}{24}\bigl[55f_3 - 59f_2 + 37f_1 - 9f_0\bigr]

    \begin{align*}
      y(0.8) &= 1.4333 + \frac{0.2}{24}\Bigl[55(1.71996) - 59(0.93880)\\

    \end{align*}
    \vfill
  \end{center}
\end{frame}

%% ── DETAILED CALCULATION ────────────────────────────────────────
\begin{frame}{\textbf{Step 3: Detailed Arithmetic}}
  \begin{center}
    \vfill
    \textbf{Evaluating term by term:}
    \begin{align*}
      55 \times 1.71996 &= 94.5978 \\
      59 \times 0.93880 &= 55.3892 \\
      37 \times 0.41632 &= 15.4038 \\
       9 \times 0.00000 &= 0.0000
    \end{align*}

    \vspace{0.5em}
    \textbf{Sum of bracketed expression:}
    \[
      94.5978 - 55.3892 + 15.4038 - 0 \;=\; 54.6124
    \]

    \vspace{0.5em}
    \textbf{Final computation:}
    \[
      y(0.8) = 1.4333 + \frac{0.2}{24} \times 54.6124

    \]

%% ── EXISTENCE & UNIQUENESS ──────────────────────────────────────
\begin{frame}{\textbf{Advanced: Existence \& Uniqueness (Picard--Lindel\"of)}}
  \begin{center}
    \vfill
    The \textbf{Picard--Lindel\"of Theorem} guarantees a \textbf{unique solution}\\[0.3em]
    exists for the IVP $y' = f(x,y)$, $y(x_0) = y_0$ if:

    \vspace{1em}
    \begin{enumerate}
      \item $f(x,y) = 2xy$ is \textbf{continuous} on $\mathbb{R}^2$ \;\textcolor{okgreen}{\checkmark}
      \item $\dfrac{\partial f}{\partial y} = 2x$ is \textbf{continuous} (Lipschitz condition) \;\textcolor{okgreen}{\checkmark}
    \end{enumerate}

    \vspace{1.5em}
    \textbf{Therefore:}\; the solution $y(x) = e^{x^2}$ is the \textit{unique}\\[0.3em]
    solution to this IVP on any closed interval containing $x = 0$.

    \vspace{1em}
    \textit{The Lipschitz constant on $[0, 0.8]$ is $L = 2 \times 0.8 = 1.6$.}
    \vfill
  \end{center}
\end{frame}

%% ── STABILITY ANALYSIS ─────────────────────────────────────────
\begin{frame}{\textbf{Advanced: Stability of the AB4 Method}}
  \begin{center}
    \vfill
    The AB4 method applied to the test equation $y' = \lambda y$ yields\\[0.3em]
    the \textbf{characteristic polynomial}:
    \[
      \rho(\zeta) - \bar{h}\,\sigma(\zeta) = 0
    \]

    \vspace{0.5em}
    where $\bar{h} = h\lambda$ and:
    \begin{align*}
      \rho(\zeta) &= \zeta^4 - \zeta^3 \\
      \sigma(\zeta) &= \tfrac{1}{24}\bigl(55\zeta^3 - 59\zeta^2 + 37\zeta - 9\bigr)
    \end{align*}

    \vspace{1em}
    The AB4 method is \textbf{zero-stable} $(\rho(\zeta)$ has roots inside \\
    the unit circle with simple roots on the boundary$)$.

    \vspace{1em}
    \textbf{Stability interval:}\; $\bar{h} = h\lambda \in [-0.3, 0]$\\[0.3em]
    For our problem: $h\lambda_{\max} = 0.2 \times 2(0.8)(1.8965) \approx 0.607$\\[0.3em]
    \textit{(marginally outside the strict stability region, but still yields\\
    accurate predictions for a single step).}
    \vfill
  \end{center}
\end{frame}

%% ── TRUNCATION ERROR ────────────────────────────────────────────
\begin{frame}{\textbf{Advanced: Local Truncation Error Analysis}}
  \begin{center}
    \vfill
    The \textbf{local truncation error} (LTE) of the AB4 method is:
    \[
      \tau_{n+1} \;=\; \frac{251}{720}\,h^5\,y^{(5)}(\xi)
      \quad\text{for some } \xi \in [x_{n-3},\,x_{n+1}]
    \]

    \vspace{1em}
    For $y = e^{x^2}$, the 5th derivative involves polynomials in $x$:
    \[
      y^{(5)}(x) \;=\; \bigl(32x^5 + 160x^3 + 120x\bigr)\,e^{x^2}
    \]

    \vspace{0.8em}
    At $x = 0.8$:\quad $y^{(5)}(0.8) \approx 153.85$

    \vspace{1em}
    \textbf{Estimated LTE:}
    \[
      |\tau| \;\approx\; \frac{251}{720} \times 0.2^5 \times 153.85
             \;\approx\; 0.0017
    \]

    \vspace{0.8em}
    \textit{This is consistent with our observed error of $0.0081$\\
    (accumulated from starting value approximations).}
    \vfill
  \end{center}
\end{frame}

%% ── PREDICTOR-CORRECTOR ────────────────────────────────────────
\begin{frame}{\textbf{Advanced: Adams--Moulton Corrector (ABM Method)}}
  \begin{center}
    \vfill
    In practice, the AB4 \textbf{predictor} is paired with the\\ \textbf{Adams--Moulton corrector} to improve accuracy:

    \vspace{1em}
    \textbf{Predict} (AB4):
    \[
      y_{n+1}^P = y_n + \frac{h}{24}\bigl[55f_n - 59f_{n-1} + 37f_{n-2} - 9f_{n-3}\bigr]
    \]

    \vspace{0.8em}
    \textbf{Correct} (AM3):
    \[
      y_{n+1}^C = y_n + \frac{h}{24}\bigl[9f_{n+1}^P + 19f_n - 5f_{n-1} + f_{n-2}\bigr]
    \]

    \vspace{1em}
    where $f_{n+1}^P = f(x_{n+1}, y_{n+1}^P)$.

    \vspace{1em}
    \textit{The corrector uses the \textbf{implicit} Adams--Moulton formula,\\
    evaluated at the predicted value -- no iteration needed!}
    \vfill
  \end{center}
\end{frame}

%% ── RK4 THEORY ──────────────────────────────────────────────────
\begin{frame}{\textbf{Numerical Verification: RK4 Method}}
  \begin{center}
    \vfill
    \textbf{Why verify with RK4?}\\
    The Runge--Kutta 4th-order method is a \textit{single-step} method\\[0.3em]
    with accuracy $\mathcal{O}(h^4)$ -- independent of AB4.

    \vspace{1em}
    For $\dfrac{dy}{dx} = f(x, y) = 2xy$:
    \begin{align*}
      k_1 &= f(x_n,\; y_n) \\
      k_2 &= f\!\left(x_n + \tfrac{h}{2},\; y_n + \tfrac{h}{2}k_1\right) \\
      k_3 &= f\!\left(x_n + \tfrac{h}{2},\; y_n + \tfrac{h}{2}k_2\right) \\
      k_4 &= f(x_n + h,\; y_n + h\,k_3)
    \end{align*}

    \vspace{0.5em}
    \[
      y_{n+1} = y_n + \frac{h}{6}(k_1 + 2k_2 + 2k_3 + k_4)
    \]
    \vfill
  \end{center}
\end{frame}

import matplotlib.pyplot as plt

# IVP: dy/dx = f(x,y) = 2xy,  y(0) = 1
def f(x, y):
    return 2 * x * y

# Given data points (step size h = 0.2)
h = 0.2
x_vals = np.array([0.0, 0.2, 0.4, 0.6])
y_vals = np.array([1.0, 1.0408, 1.1735, 1.4333])

# Compute f_i = f(x_i, y_i)
f_vals = np.array([f(x_vals[i], y_vals[i]) for i in range(4)])

# Adams-Bashforth 4-step predictor
# y_{n+1} = y_n + (h/24)[55*f_n - 59*f_{n-1}
#           + 37*f_{n-2} - 9*f_{n-3}]
y_pred = y_vals[3] + (h / 24) * (
    55*f_vals[3] - 59*f_vals[2]
    + 37*f_vals[1] - 9*f_vals[0]
)
print(f"Predicted y(0.8) = {y_pred:.4f}")
# Output: Predicted y(0.8) = 1.8884

# Exact solution: y = exp(x^2)
x_exact = np.linspace(0, 1.0, 500)
y_exact = np.exp(x_exact**2)

# Plot
x_ab = np.append(x_vals, 0.8)
y_ab = np.append(y_vals, y_pred)

plt.figure(figsize=(8, 5))
plt.plot(x_exact, y_exact, 'b-', lw=2,

import numpy as np
import matplotlib.pyplot as plt

def f(x, y):
    return 2 * x * y

# RK4 solver
h     = 0.01
x_end = 1.2
N     = int(x_end / h)

x_rk = np.zeros(N + 1)
y_rk = np.zeros(N + 1)
y_rk[0] = 1.0

for n in range(N):
    k1 = f(x_rk[n],       y_rk[n])
    k2 = f(x_rk[n]+h/2,   y_rk[n]+h/2*k1)
    k3 = f(x_rk[n]+h/2,   y_rk[n]+h/2*k2)
    k4 = f(x_rk[n]+h,     y_rk[n]+h*k3)
    y_rk[n+1] = y_rk[n] + (h/6)*(k1
                + 2*k2 + 2*k3 + k4)
    x_rk[n+1] = x_rk[n] + h

# Exact solution
x_ex = np.linspace(0, x_end, 1000)
y_ex = np.exp(x_ex**2)

fig, (ax1, ax2) = plt.subplots(1, 2,
                                figsize=(13,5))
ax1.plot(x_ex, y_ex, 'b-', lw=2,
         label=r'Exact: $y = e^{x^2}$')
ax1.plot(x_rk, y_rk, 'g--', lw=1.5,
         label='RK4 (h = 0.01)')
ax1.legend()
ax1.grid(True, alpha=0.3)

error = np.abs(y_rk - np.exp(x_rk**2))
ax2.semilogy(x_rk, error, 'r-', lw=1.5)
ax2.set_title('RK4 Absolute Error')
ax2.grid(True, alpha=0.3, which='both')
plt.tight_layout()
plt.savefig('figs/rk4_verification.png', dpi=150)
plt.show()
\end{lstlisting}
