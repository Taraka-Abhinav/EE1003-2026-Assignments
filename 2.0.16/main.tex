%% ── QUESTION SLIDE ───────────────────────────────────────
\begin{frame}{Problem Statement}
Four jobs have to be sequenced on a single facility, with the objective of minimizing
the maximum tardiness
\begin{center}
    \[(\max_i |Completion Time_i - Due Date_i|)\]
\end{center}
The jobs have due dates and processing times as follows:
\begin{table}[]
\centering
\begin{tabular}{|c|c|c|}
\hline
\textbf{Job} & \textbf{Due Date ($d_i$)} & \textbf{Processing Time ($p_i$)} \\
\hline
P & 5 & 2 \\
Q & 6 & 10 \\
R & 3 & 3 \\
S & 7 & 4 \\
\hline
\end{tabular}
\end{table}

\vspace{0.3cm}
The \textbf{last} job that should be taken up is
\begin{columns}[T]
\begin{column}{0.48\textwidth}
\begin{itemize}
    \item[(A)] P
    \item[(B)] Q
\end{itemize}
\end{column}
\begin{column}{0.48\textwidth}
\begin{itemize}
    \item[(C)] R
    \item[(D)] S
\end{itemize}
\end{column}
\end{columns}
\end{frame}

%% ── UNDERSTANDING THE OBJECTIVE ──────────────────────────
\begin{frame}{Understanding the Objective}
We need to minimize the maximum absolute deviation from due dates:
\[
\min \left( \max_i |C_i - d_i| \right)
\]

\textbf{Key Definitions:}
\begin{itemize}
\item \textbf{Completion time ($C_i$):} Time at which job $i$ finishes processing.
\item \textbf{Lateness ($L_i$):} $L_i = C_i - d_i$
\item \textbf{Absolute Lateness:} $|L_i| = |C_i - d_i|$
\end{itemize}

\vspace{0.2cm}
\textbf{Observations:}
\begin{itemize}
\item If $C_i > d_i \Rightarrow$ job is \textcolor{darkred}{Late}
\item If $C_i < d_i \Rightarrow$ job is \textcolor{okgreen}{Early}
\item The objective behaves exactly like minimizing the \textbf{Maximum Lateness} ($L_{\max}$).
\end{itemize}
\end{frame}

%% ── OTHER HEURISTICS (NEW SLIDE) ─────────────────────────
\begin{frame}{Alternative Scheduling Methods}
In operations research, several heuristic rules are taught for single-machine scheduling. Why don't we use them here?

\vspace{0.2cm}
\begin{itemize}
    \item \textbf{SPT (Shortest Processing Time):} Sort by increasing $p_i$.
    \begin{itemize}
        \item \textit{Pros:} Minimizes average completion time (mean flow time).
        \item \textit{Cons:} Ignores due dates entirely. Bad for lateness.
    \end{itemize}
    \vspace{0.2cm}
    \item \textbf{LST (Least Slack Time):} Sort by increasing $(d_i - p_i)$.
    \begin{itemize}
        \item \textit{Pros:} Good for minimizing maximum tardiness in some scenarios.
        \item \textit{Cons:} Does not guarantee optimal $L_{\max}$.
    \end{itemize}
    \vspace{0.2cm}
    \item \textbf{EDD (Earliest Due Date):} Sort by increasing $d_i$.
    \begin{itemize}
        \item \textit{Pros:} Mathematically proven to minimize maximum lateness ($L_{\max}$).
    \end{itemize}
\end{itemize}
\end{frame}

%% ── SPT COMPARISON (NEW SLIDE) ───────────────────────────
\begin{frame}{Comparison: Why SPT Fails Here}
Let's try the \textbf{SPT (Shortest Processing Time)} rule:
Sequence by smallest $p_i$: \textbf{P(2) $\rightarrow$ R(3) $\rightarrow$ S(4) $\rightarrow$ Q(10)}

\vspace{0.3cm}
\begin{table}[]
\centering
\renewcommand{\arraystretch}{1.2}
\begin{tabular}{|c|c|c|c|}
\hline
\textbf{Job} & \textbf{$C_i$ (Cumulative)} & \textbf{$d_i$} & \textbf{Absolute ($|C_i - d_i|$)} \\
\hline
P & 2 & 5 & $|2 - 5| = 3$ \\
R & 2+3 = 5 & 3 & $|5 - 3| = 2$ \\
S & 5+4 = 9 & 7 & $|9 - 7| = 2$ \\
Q & 9+10 = 19 & 6 & $|19 - 6| = \res{13}$ \\
\hline
\end{tabular}
\end{table}

\vspace{0.3cm}
\[
\text{Maximum Deviation using SPT} = \res{13}
\]

Because Q takes 10 units but is due at 6, pushing it to the end creates a massive delay. We need a rule that respects due dates!
\end{frame}

%% ── THEORY RULE (EDD) ────────────────────────────────────
\begin{frame}{The Optimal Rule: EDD}
For minimizing maximum absolute deviation ($L_{\max}$), we must use Jackson's Theorem:

\vspace{0.2cm}
\begin{center}
\textbf{EDD Rule (Earliest Due Date First)}
\end{center}

\vspace{0.4cm}
\textbf{Algorithm Idea:}
\begin{itemize}
\item \textbf{Sorting:} Jobs are sequenced in increasing order of their due dates ($d_i$).
\item \textbf{Complexity:} Runs in $\mathcal{O}(n \log n)$ time.
\end{itemize}

\vspace{0.2cm}
\textbf{Important Result:}
\[
\boxed{\text{EDD guarantees the minimum possible } L_{\max}}
\]
\end{frame}

%% ── STEP 1 ───────────────────────────────────────────────
\begin{frame}{Step 1: Apply EDD to Given Data}
Let's look at the given jobs and their due dates:

\vspace{0.2cm}
\begin{table}[]
\centering
\begin{tabular}{|c|c|c|}
\hline
\textbf{Job} & \textbf{Due Date ($d_i$)} & \textbf{Processing Time ($p_i$)} \\
\hline
P & 5 & 2 \\
Q & 6 & 10 \\
R & 3 & 3 \\
S & 7 & 4 \\
\hline
\end{tabular}
\end{table}

\vspace{0.3cm}
Sort by increasing due date ($d_i$):
\[
R \,(d_R=3), \quad P \,(d_P=5), \quad Q \,(d_Q=6), \quad S \,(d_S=7)
\]

\vspace{0.2cm}
\[
\Rightarrow \text{Optimal Sequence: } R \rightarrow P \rightarrow Q \rightarrow S
\]
\end{frame}

%% ── STEP 2 ───────────────────────────────────────────────
\begin{frame}{Step 2: Compute Completion Times}
\textbf{Cumulative Calculation:}

Since jobs are processed one after another without idle time, the completion time ($C_i$) is the sum of processing times.

\vspace{0.3cm}
\[
\begin{aligned}
C_R &= p_R = \res{3} \\
C_P &= C_R + p_P = 3 + 2 = \res{5} \\
C_Q &= C_P + p_Q = 5 + 10 = \res{15} \\
C_S &= C_Q + p_S = 15 + 4 = \res{19}
\end{aligned}
\]

\vspace{0.3cm}
\textbf{Note:} Total time taken to finish all jobs (Makespan) is 19 units.
\end{frame}

%% ── STEP 3 ───────────────────────────────────────────────
\begin{frame}{Step 3: Compute Lateness and Absolute Values}
Using the formula: $L_i = C_i - d_i$

\vspace{0.2cm}
\begin{table}[]
\centering
\renewcommand{\arraystretch}{1.2}
\begin{tabular}{|c|c|c|c|c|}
\hline
\textbf{Job} & \textbf{$C_i$} & \textbf{$d_i$} & \textbf{Lateness ($L_i$)} & \textbf{Absolute ($|C_i - d_i|$)} \\
\hline
R & 3 & 3 & 0 & \res{0} \\
P & 5 & 5 & 0 & \res{0} \\
Q & 15 & 6 & +9 & \res{9} \\
S & 19 & 7 & +12 & \res{12} \\
\hline
\end{tabular}
\end{table}

\vspace{0.4cm}
Evaluating the Maximum Deviation:
\[
\max_i |C_i - d_i| = \max(0, 0, 9, 12) = \res{12}
\]
The maximum lateness in this optimal sequence is 12 (Better than SPT's 13).
\end{frame}

%% ── FINAL CONCLUSION ─────────────────────────────────────
\begin{frame}{Final Conclusion}
The optimal sequence using Earliest Due Date (EDD) is:
\[
R \rightarrow P \rightarrow Q \rightarrow S
\]

\vspace{0.4cm}
The question asks for the \textbf{last job} executed in the optimal schedule.
\[
\text{Last Job } = \boxed{\text{Job } S}
\]

\vspace{0.4cm}
\textbf{Final Answer:}
\[
\boxed{\textbf{(D) S}}
\]
\end{frame}

\begin{frame}[fragile]{Python: Data Setup \& Analysis}
\begin{lstlisting}
import matplotlib.pyplot as plt
import numpy as np

# Job Data
jobs = {
    'P': {'p': 2, 'd': 5},
    'Q': {'p': 10, 'd': 6},
    'R': {'p': 3, 'd': 3},
    'S': {'p': 4, 'd': 7}
}

# Sequences
seq_edd = ['R', 'P', 'Q', 'S']
seq_spt = ['P', 'R', 'S', 'Q']
seq_lst = ['Q', 'R', 'S', 'P']

def analyze_sequence(seq):
    current_time = 0
    gantt = []
    abs_lateness = {}
    actual_lateness = {}
    cumulative_times = [0]
\end{lstlisting}
\end{frame}
\begin{frame}[fragile]{Python: Data Setup \& Analysis}
\begin{lstlisting}
    for job in seq:
        p = jobs[job]['p']
        d = jobs[job]['d']
        start = current_time
        current_time += p

        gantt.append((job, start, p, d))
        abs_lateness[job] = abs(current_time - d)
        actual_lateness[job] = current_time - d
        cumulative_times.append(current_time)

    return gantt, abs_lateness, actual_lateness, cumulative_times
\end{lstlisting}
\end{frame}

\begin{frame}[fragile]{Python: Gantt Chart Generation}

draw_gantt_row(ax1, gantt_lst, 5)
