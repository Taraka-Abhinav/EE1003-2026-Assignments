\section{Question}

\begin{frame}{Question: The Problem Statement}
\textbf{Evaluate the double integral:}

\[
I = \int_0^\infty \int_0^\infty e^{-x^2 - y^2} \, dx\,dy
\]

\vspace{0.5cm}

\textbf{Given:}
\begin{itemize}
    \item Region of integration: \textbf{First Quadrant} $(x \ge 0, y \ge 0)$
    \item Integrand: $e^{-x^2 - y^2}$ (always positive and symmetric)
    \item Find the exact value of this integral
\end{itemize}

\vspace{0.3cm}

\textbf{Source:} IN 2007, Problem 3.0.16
\end{frame}

% ══════════════════════════════════════════════════════════════════

\section{Final Result}

% ── FINAL ANSWER ──────────────────────────────────────────────────
\begin{frame}{Final Answer}
\begin{center}
\LARGE
\[
\boxed{
\int_0^\infty \int_0^\infty e^{-x^2 - y^2} \, dx\,dy
= \res{\frac{\pi}{4}}
}
\]
\end{center}

\vspace{0.5cm}

\begin{itemize}
    \item \textbf{Exact value:} $\pi/4$
    \item \textbf{Approximate value:} $0.785398...$
    \item \textbf{Methods verified:} 5 (Matrix, Polar, Separation, Error Function, Monte Carlo)
    \item \textbf{Python visualizations:} 6 plots generated
\end{itemize}
\end{frame}

import numpy as np
import matplotlib.pyplot as plt

def monte_carlo_integral(N, L=10):
    x = np.random.uniform(0, L, N)
    y = np.random.uniform(0, L, N)
    f_values = np.exp(-x**2 - y**2)
    return L**2 * np.mean(f_values)

# Convergence analysis
sample_sizes = np.logspace(2, 6, 50, dtype=int)
estimates = [monte_carlo_integral(N) for N in sample_sizes]

plt.plot(sample_sizes, estimates, 'b-', linewidth=2)

            label=r'True value: $\pi/4$')

\section{Solution Methods}

% ── METHOD 0: MATRIX FORMULATION ──────────────────────────────────
\begin{frame}{Solution - Method 0: Matrix Formulation}
\textbf{Step 1: Matrix Representation}
\[
x^2 + y^2 =
\begin{bmatrix} x & y \end{bmatrix}
\begin{bmatrix} 1 & 0 \\ 0 & 1 \end{bmatrix}
\begin{bmatrix} x \\ y \end{bmatrix}
= \bm{x}^T A \bm{x}
\]

where $A = I$ (Identity matrix), symmetric and positive definite.

\vspace{0.3cm}

\textbf{Step 2: Standard Gaussian Result}
\[
\int_{\mathbb{R}^n} e^{-\bm{x}^T A \bm{x}} d\bm{x}
= \sqrt{\frac{\pi^n}{\det(A)}}
\]

For $n = 2$, $A = I$, $\det(A) = 1$:
\[
\int_{-\infty}^{\infty} \int_{-\infty}^{\infty}
e^{-(x^2 + y^2)} dx\,dy = \pi
\]
\end{frame}

\begin{frame}{Solution - Method 0: Matrix Formulation (cont.)}
\textbf{Step 3: Restricting to First Quadrant}

\begin{itemize}
    \item Given region: $(x \ge 0, y \ge 0)$ is \textbf{one-fourth} of full plane
    \item Function is symmetric in all quadrants
\end{itemize}

\[
I = \frac{1}{4} \times \pi = \res{\frac{\pi}{4}}
\]

\vspace{0.5cm}

\textbf{Result:} $\boxed{I = \frac{\pi}{4} \approx 0.7854}$
\end{frame}

% ── METHOD 1: POLAR COORDINATES ───────────────────────────────────
\begin{frame}{Solution - Method 1: Polar Coordinates}
\textbf{Key Idea:} Convert Cartesian to polar coordinates

\textbf{Transformation:}
\[
x = r\cos\theta, \quad y = r\sin\theta, \quad dA = r\,dr\,d\theta
\]
\[
x^2 + y^2 = r^2
\]

\textbf{Integral becomes:}
\[
I = \int_0^{\pi/2} \int_0^\infty e^{-r^2} r \, dr\,d\theta
\]

\textbf{Step 1: Separate the integrals}
\[
I = \left(\int_0^{\pi/2} d\theta\right) \left(\int_0^\infty r e^{-r^2} dr\right)
\]
\end{frame}

\begin{frame}{Solution - Method 1: Polar Coordinates (cont.)}
\textbf{Step 2: Evaluate angular part}
\[
\int_0^{\pi/2} d\theta = \frac{\pi}{2}
\]

\textbf{Step 3: Evaluate radial part}

Substitution: $u = r^2$, $du = 2r\,dr$
\[
\int_0^\infty r e^{-r^2} dr = \frac{1}{2}\int_0^\infty e^{-u} du = \frac{1}{2}
\]

\textbf{Result:}
\[
I = \frac{\pi}{2} \times \frac{1}{2} = \res{\frac{\pi}{4}}
\]
\end{frame}

% ── METHOD 2: SEPARATION OF VARIABLES ─────────────────────────────
\begin{frame}{Solution - Method 2: Separation of Variables}
\textbf{Key Idea:} Express as product of independent integrals

\textbf{Observation:}
\[
e^{-x^2 - y^2} = e^{-x^2} \cdot e^{-y^2}
\]

\textbf{Factorization:}
\[
I = \int_0^\infty \int_0^\infty e^{-x^2} e^{-y^2} \, dx\,dy
\]
\[
= \left(\int_0^\infty e^{-x^2} dx\right) \left(\int_0^\infty e^{-y^2} dy\right)
\]

Since both integrals are identical:
\[
I = \left(\int_0^\infty e^{-x^2} dx\right)^2
\]
\end{frame}

\begin{frame}{Solution - Method 2: Separation of Variables (cont.)}
\textbf{Standard Gaussian Integral:}
\[
\int_0^\infty e^{-x^2} dx = \frac{\sqrt{\pi}}{2}
\]

\textbf{Derivation:} Let $I_1 = \int_0^\infty e^{-x^2} dx$
\[
I_1^2 = \left(\int_0^\infty e^{-x^2} dx\right)\left(\int_0^\infty e^{-y^2} dy\right) = \int_0^\infty \int_0^\infty e^{-(x^2+y^2)} dx\,dy
\]

Converting to polar: $I_1^2 = \frac{\pi}{4}$, thus $I_1 = \frac{\sqrt{\pi}}{2}$

\vspace{0.3cm}

\textbf{Final Result:}
\[
I = \left(\frac{\sqrt{\pi}}{2}\right)^2 = \res{\frac{\pi}{4}}
\]
\end{frame}

% ── METHOD 3: ERROR FUNCTION ──────────────────────────────────────
\begin{frame}{Solution - Method 3: Error Function Approach}
\textbf{Error Function Definition:}
\[
\text{erf}(x) = \frac{2}{\sqrt{\pi}} \int_0^x e^{-t^2} dt
\]

\textbf{Key Property:}
\[
\lim_{x \to \infty} \text{erf}(x) = 1 \quad \Rightarrow \quad \int_0^\infty e^{-t^2} dt = \frac{\sqrt{\pi}}{2}
\]

\textbf{Express integral using erf:}
\[
I = \lim_{a,b \to \infty} \int_0^a \int_0^b e^{-x^2-y^2} dx\,dy
\]
\[
= \lim_{a,b \to \infty} \left(\int_0^a e^{-x^2} dx\right)\left(\int_0^b e^{-y^2} dy\right)
\]
\[
= \lim_{a,b \to \infty} \frac{\sqrt{\pi}}{2}\text{erf}(a) \cdot \frac{\sqrt{\pi}}{2}\text{erf}(b) = \res{\frac{\pi}{4}}
\]
\end{frame}

% ── METHOD 4: MONTE CARLO ─────────────────────────────────────────
\begin{frame}{Solution - Method 4: Monte Carlo Numerical Integration}
\textbf{Key Idea:} Use random sampling to approximate the integral

\textbf{Monte Carlo Method:}
\begin{itemize}
    \item Generate random points $(x_i, y_i)$ in region $[0, L] \times [0, L]$
    \item Evaluate $f(x_i, y_i) = e^{-x_i^2 - y_i^2}$ at each point
    \item Average the function values and multiply by area
\end{itemize}

\textbf{Formula:}
\[
I \approx L^2 \times \frac{1}{N} \sum_{i=1}^N e^{-x_i^2 - y_i^2}
\]

For large $L$ and $N \to \infty$:
\[
I \approx \res{\frac{\pi}{4}} \approx 0.7854
\]
\end{frame}
