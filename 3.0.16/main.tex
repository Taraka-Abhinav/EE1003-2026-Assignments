% ── SLIDE 2: THE PROBLEM ──────────────────────────────────────────
\begin{frame}{1. The Problem Statement}
\[
I = \int_0^\infty \int_0^\infty e^{-x^2 - y^2} \, dx\,dy
\]

\begin{itemize}
    \item Evaluate the given Gaussian double integral.
    \item Region of integration: \textbf{First Quadrant}
    \item Integrand is always \textbf{positive and symmetric}
\end{itemize}
\end{frame}

% ── SLIDE 3: MATRIX FORMULATION ───────────────────────────────────
\begin{frame}{2. Matrix Representation}
\[
x^2 + y^2 =
\begin{bmatrix} x & y \end{bmatrix}
\begin{bmatrix} 1 & 0 \\ 0 & 1 \end{bmatrix}
\begin{bmatrix} x \\ y \end{bmatrix}
= \bm{x}^T A \bm{x}
\]

\begin{itemize}
    \item \( A = I \) (Identity matrix)
    \item Symmetric and Positive Definite
\end{itemize}

\[
I = \int_{x \ge 0, y \ge 0} e^{-\bm{x}^T A \bm{x}} \, d\bm{x}
\]
\end{frame}

% ── SLIDE 4: STANDARD RESULT ──────────────────────────────────────
\begin{frame}{3. Standard Gaussian Result}
\[
\int_{\mathbb{R}^n} e^{-\bm{x}^T A \bm{x}} d\bm{x}
= \sqrt{\frac{\pi^n}{\det(A)}}
\]

For this case:
\[
n = 2, \quad A = I, \quad \det(A) = 1
\]

\[
\Rightarrow \int_{-\infty}^{\infty} \int_{-\infty}^{\infty}
e^{-(x^2 + y^2)} dx\,dy = \res{\pi}
\]
\end{frame}

% ── SLIDE 5: REGION ARGUMENT ──────────────────────────────────────
\begin{frame}{4. Restricting the Region}
\begin{itemize}
    \item Given region: \( x \ge 0, y \ge 0 \)
    \item This is \textbf{one-fourth} of the full plane
    \item Function is symmetric in all quadrants
\end{itemize}

\[
I = \frac{1}{4} \times \pi = \res{\frac{\pi}{4}}
\]
\end{frame}

% ── SLIDE 6: FINAL RESULT ─────────────────────────────────────────
\begin{frame}{5. Final Answer}
\[
\boxed{
\int_0^\infty \int_0^\infty e^{-x^2 - y^2} \, dx\,dy
= \res{\frac{\pi}{4}}
}
\]

\vspace{0.5cm}

\begin{itemize}
    \item Exact value: \( \frac{\pi}{4} \)
    \item Approximate value: \( 0.7854 \)
\end{itemize}
\end{frame}

% ── NEW: METHOD 1 - POLAR COORDINATES ─────────────────────────────
\begin{frame}{Method 1: Polar Coordinates (Classical Approach)}
\textbf{Key Idea:} Convert Cartesian to polar coordinates using symmetry

\vspace{0.3cm}

\textbf{Transformation:}
\[
x = r\cos\theta, \quad y = r\sin\theta, \quad dA = r\,dr\,d\theta
\]

\[
x^2 + y^2 = r^2
\]

\textbf{Integral becomes:}
\[
I = \int_0^{\pi/2} \int_0^\infty e^{-r^2} r \, dr\,d\theta
\]
\end{frame}

\begin{frame}{Method 1: Polar Coordinates (Solution)}
\textbf{Step 1: Separate the integrals}
\[
I = \left(\int_0^{\pi/2} d\theta\right) \left(\int_0^\infty r e^{-r^2} dr\right)
\]

\textbf{Step 2: Evaluate angular part}
\[
\int_0^{\pi/2} d\theta = \frac{\pi}{2}
\]

\textbf{Step 3: Evaluate radial part (substitution: $u = r^2$)}
\[
\int_0^\infty r e^{-r^2} dr = \frac{1}{2}\int_0^\infty e^{-u} du = \frac{1}{2}
\]

\textbf{Result:}
\[
I = \frac{\pi}{2} \times \frac{1}{2} = \res{\frac{\pi}{4}}
\]
\end{frame}

% ── NEW: METHOD 2 - SEPARATION OF VARIABLES ───────────────────────
\begin{frame}{Method 2: Separation of Variables}
\textbf{Key Idea:} Express as product of independent integrals

\vspace{0.3cm}

\textbf{Observation:}
\[
e^{-x^2 - y^2} = e^{-x^2} \cdot e^{-y^2}
\]

\textbf{Factorization:}
\[
I = \int_0^\infty \int_0^\infty e^{-x^2} e^{-y^2} \, dx\,dy = \left(\int_0^\infty e^{-x^2} dx\right) \left(\int_0^\infty e^{-y^2} dy\right)
\]

\textbf{Since both integrals are identical:}
\[
I = \left(\int_0^\infty e^{-x^2} dx\right)^2
\]
\end{frame}

\begin{frame}{Method 2: Gaussian Integral Evaluation}
\textbf{Standard Gaussian Integral:}
\[
\int_0^\infty e^{-x^2} dx = \frac{\sqrt{\pi}}{2}
\]

\textbf{Derivation uses polar trick:} Let $I_1 = \int_0^\infty e^{-x^2} dx$
\[
I_1^2 = \left(\int_0^\infty e^{-x^2} dx\right)\left(\int_0^\infty e^{-y^2} dy\right) = \int_0^\infty \int_0^\infty e^{-(x^2+y^2)} dx\,dy
\]

Converting to polar: $I_1^2 = \frac{\pi}{4}$, thus $I_1 = \frac{\sqrt{\pi}}{2}$

\vspace{0.3cm}

\textbf{Final Result:}
\[
I = \left(\frac{\sqrt{\pi}}{2}\right)^2 = \res{\frac{\pi}{4}}
\]
\end{frame}

% ── NEW: METHOD 3 - ERROR FUNCTION APPROACH ───────────────────────
\begin{frame}{Method 3: Error Function Approach}
\textbf{Error Function Definition:}
\[
\text{erf}(x) = \frac{2}{\sqrt{\pi}} \int_0^x e^{-t^2} dt
\]

\textbf{Key Property:}
\[
\lim_{x \to \infty} \text{erf}(x) = 1 \quad \Rightarrow \quad \int_0^\infty e^{-t^2} dt = \frac{\sqrt{\pi}}{2}
\]

\textbf{Express integral using erf:}
\[
I = \lim_{a,b \to \infty} \int_0^a \int_0^b e^{-x^2-y^2} dx\,dy
\]
\[
= \lim_{a,b \to \infty} \left(\int_0^a e^{-x^2} dx\right)\left(\int_0^b e^{-y^2} dy\right)
\]
\[
= \lim_{a,b \to \infty} \frac{\sqrt{\pi}}{2}\text{erf}(a) \cdot \frac{\sqrt{\pi}}{2}\text{erf}(b) = \res{\frac{\pi}{4}}
\]
\end{frame}

% ── NEW: METHOD 4 - NUMERICAL INTEGRATION (MONTE CARLO) ───────────
\begin{frame}{Method 4: Numerical Integration (Monte Carlo)}
\textbf{Key Idea:} Use random sampling to approximate the integral

\vspace{0.3cm}

\textbf{Monte Carlo Method:}
\begin{itemize}
    \item Generate random points $(x_i, y_i)$ in region $[0, L] \times [0, L]$
    \item Evaluate $f(x_i, y_i) = e^{-x_i^2 - y_i^2}$ at each point
    \item Average the function values and multiply by area
\end{itemize}

\textbf{Formula:}
\[
I \approx L^2 \times \frac{1}{N} \sum_{i=1}^N e^{-x_i^2 - y_i^2}
\]

For large $L$ and $N \to \infty$:
\[
I \approx \res{\frac{\pi}{4}} \approx 0.7854
\]
\end{frame}
