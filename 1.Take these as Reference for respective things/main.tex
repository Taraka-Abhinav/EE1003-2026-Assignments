%% ── SOLUTION 1: Initial Condition ────────────────────────────────
\begin{frame}{\textbf{Initial Condition}}
  \begin{center}
    \vfill
    \textbf{Given:}\quad $V_C(t=0^-) = 0\,\mathrm{V}$

    \vspace{1.2em}
    A capacitor \textbf{cannot change its voltage instantaneously}\\[0.4em]
    because an instantaneous change requires \textbf{infinite current},\\[0.4em]
    which is physically impossible.

    \vspace{1.5em}
    Therefore:
    \[
      V_C(t=0^+) \;=\; V_C(t=0^-) \;=\; \mathbf{0\,V}
    \]

    \vspace{0.6em}
    At $t = 0^+$, since $V_C = 0$, the capacitor behaves as a \textbf{short circuit}.
    \vfill
  \end{center}
\end{frame}

%% ── SOLUTION 2: Steady-State Voltage ─────────────────────────────
\begin{frame}{\textbf{Steady-State Voltage $V_\infty$}}
  \begin{center}
    \vfill
    As $t \to \infty$, the capacitor is \textbf{fully charged}\\[0.4em]
    \rarrow no current flows into it \rarrow capacitor is an \textbf{open circuit}

    \vspace{1.2em}
    The two resistors then form a simple \textbf{Voltage Divider}:

    \vspace{0.8em}
    \[
      V_\infty \;=\; V_s \times \frac{R_2}{R_1 + R_2}
              \;=\; 10 \times \frac{20\,\mathrm{k}}{20\,\mathrm{k} + 20\,\mathrm{k}}
              \;=\; 10 \times \frac{1}{2}
    \]

    \vspace{1em}
    \Large\textbf{\textcolor{madblue}{$V_\infty \;=\; 5\,\mathrm{V}$}}
    \vfill
  \end{center}
\end{frame}

%% ── SOLUTION 3: Thevenin Resistance ──────────────────────────────
\begin{frame}{\textbf{Thévenin Resistance $R_{th}$}}
  \begin{center}
    \vfill
    \textbf{Deactivate} the independent voltage source\\[0.3em]
    (replace with a \textbf{short circuit})

    \vspace{1em}
    Looking from the capacitor terminals, both $20\,\mathrm{k}\Omega$ resistors\\[0.3em]
    are now connected \textbf{in parallel}:

    \vspace{0.8em}
    \[
      R_{th} \;=\; 20\,\mathrm{k} \;\|\; 20\,\mathrm{k}
             \;=\; \frac{20 \times 20}{20 + 20}\,\mathrm{k}
             \;=\; \frac{400}{40}\,\mathrm{k}
    \]

    \vspace{1em}
    \Large\textbf{\textcolor{madblue}{$R_{th} \;=\; 10\,\mathrm{k}\Omega$}}
    \vfill
  \end{center}
\end{frame}

%% ── SOLUTION 4: Time Constant ────────────────────────────────────
\begin{frame}{\textbf{Time Constant $\tau$}}
  \begin{center}
    \vfill
    The time constant for an RC circuit is:
    \[
      \tau \;=\; R_{th} \cdot C
    \]

    \vspace{0.8em}
    Substituting $R_{th} = 10\,\mathrm{k}\Omega$ and $C = 4\,\mu\mathrm{F}$:
    \[
      \tau \;=\; 10 \times 10^{3} \;\times\; 4 \times 10^{-6}
           \;=\; 40 \times 10^{-3}\,\mathrm{s}
    \]

    \vspace{0.8em}
    \large\textbf{\textcolor{madblue}{$\tau \;=\; 40\,\mathrm{ms}$}}

    \vspace{1em}
    Therefore the decay exponent:
    \[
      \frac{1}{\tau} \;=\; \frac{1}{0.04} \;=\; \mathbf{25\,s^{-1}}
    \]
    \vfill
  \end{center}
\end{frame}

%% ── SOLUTION 5: Capacitor Voltage ────────────────────────────────
\begin{frame}{\textbf{Capacitor Voltage $V_C(t)$}}
  \begin{center}
    \vfill
    The standard RC complete response formula:
    \[
      V_C(t) \;=\; V_\infty \;+\; \bigl[V_C(t=0^+) - V_\infty\bigr]\,e^{-t/\tau}
    \]

    \vspace{0.8em}
    Substituting $V_\infty = 5\,\mathrm{V}$,\quad $V_C(t=0^+) = 0\,\mathrm{V}$,\quad $1/\tau = 25$:
    \[
      V_C(t) \;=\; 5 \;+\; [0 - 5]\,e^{-25t}
    \]

    \vspace{1em}
    \Large\textbf{\textcolor{madblue}{$V_C(t) \;=\; 5\bigl(1 - e^{-25t}\bigr)\,\mathrm{V}$}}

    \vspace{1.2em}
    \normalsize
    \textbf{Check:}\quad
    At $t=0$:\; $V_C = 5(1-1) = 0\,\mathrm{V}$ \;\textcolor{okgreen}{\checkmark}
    \qquad
    As $t\to\infty$:\; $V_C \to 5\,\mathrm{V}$ \;\textcolor{okgreen}{\checkmark}
    \vfill
  \end{center}
\end{frame}

%% ── SOLUTION 6: Capacitor Current ────────────────────────────────
\begin{frame}{\textbf{Capacitor Current $i_c(t)$}}
  \begin{center}
    \vfill
    Using the fundamental capacitor relation:
    \[
      i_c(t) \;=\; C\,\frac{dV_C}{dt}
    \]

    \vspace{0.6em}
    Differentiating $V_C(t) = 5(1 - e^{-25t})$:
    \[
      \frac{dV_C}{dt} \;=\; 5 \times 25\,e^{-25t} \;=\; 125\,e^{-25t}\,\mathrm{V/s}
    \]

    \vspace{0.6em}
    Multiplying by $C = 4 \times 10^{-6}\,\mathrm{F}$:
    \[
      i_c(t) \;=\; 4\times10^{-6} \;\times\; 125\,e^{-25t}
             \;=\; 500\times10^{-6}\,e^{-25t}\,\mathrm{A}
    \]

    \vspace{1em}
    \LARGE\textbf{\textcolor{madblue}{$i_c(t) \;=\; 0.50\,e^{-25t}\,\mathrm{mA}$}}

    \vspace{0.5em}
    \normalsize\textbf{\textcolor{okgreen}{Answer: Option (a)}}
    \vfill
  \end{center}
\end{frame}

%% ── SOLUTION 7: KCL Verification ─────────────────────────────────
\begin{frame}{\textbf{Verification via KCL}}
  \begin{center}
    \vfill
    At the junction node, applying KCL:
    \[
      i_c \;=\; \underbrace{\frac{10 - V_C}{20\,\mathrm{k}}}_{i_1\text{ (series R)}}
               \;-\;
               \underbrace{\frac{V_C}{20\,\mathrm{k}}}_{i_2\text{ (shunt R)}}
             \;=\; \frac{10 - 2V_C}{20\,\mathrm{k}}
    \]

    \vspace{0.8em}
    Substituting $V_C = 5(1 - e^{-25t})$, so $\;2V_C = 10(1 - e^{-25t})$:
    \[
      i_c(t) \;=\; \frac{10 - 10(1 - e^{-25t})}{20\,\mathrm{k}}
             \;=\; \frac{10\,e^{-25t}}{20\,\mathrm{k}}
    \]

    \vspace{1em}
    \Large\textbf{\textcolor{madblue}{$i_c(t) \;=\; 0.50\,e^{-25t}\,\mathrm{mA}$}}\;
    \textcolor{okgreen}{\checkmark}
    \vfill
  \end{center}
\end{frame}

%% ── ADVANCED: ODE Formulation ────────────────────────────────────
\begin{frame}{\textbf{Advanced: ODE Formulation}}
  \begin{center}
    \vfill
    Applying \textbf{KCL} at the capacitor node for $t > 0$:
    \[
      C\frac{dV_C(t)}{dt} \;+\; \frac{V_C(t)}{R_2} \;=\; \frac{V_s - V_C(t)}{R_1}
    \]

    \vspace{1.5em}
    Rearranging into the standard first-order linear ODE form:
    \[
      \frac{dV_C(t)}{dt} \;+\; \left( \frac{R_1 + R_2}{R_1 R_2 C} \right) V_C(t)
                         \;=\; \frac{V_s}{R_1 C}
    \]

    \vspace{1.5em}
    Substituting circuit values:
    \[
      \frac{dV_C(t)}{dt} \;+\; \mathbf{25\,V_C(t)} \;=\; \mathbf{125}
    \]
    \vfill
  \end{center}
\end{frame}

%% ── ADVANCED: ODE Solution ───────────────────────────────────────
\begin{frame}{\textbf{Advanced: ODE Solution}}
  \begin{center}
    \vfill
    From the derived ODE:
    \[
      \frac{dV_C(t)}{dt} \;+\; 25\,V_C(t) \;=\; 125
    \]

    \vspace{1.5em}
    The complete response is the sum of \textbf{natural} and \textbf{forced} responses:
    \[
      V_C(t) \;=\; K e^{-25t} \;+\; 5
    \]

    \vspace{1.5em}
    Applying the initial condition $V_C(0^+) = 0$, we find $K = -5$.

    \vspace{2em}
    \Large\textbf{\textcolor{madblue}{$V_C(t) \;=\; 5\bigl(1 - e^{-25t}\bigr)\,\mathrm{V}$}}
    \vfill
  \end{center}
\end{frame}

%% ── ADVANCED: Transfer Function ──────────────────────────────────
\begin{frame}{\textbf{Advanced: Transfer Function $H(s)$}}
  \begin{center}
    \vfill
    Analyzing the circuit as a system mapping $V_s(t) \to V_C(t)$ in the $s$-domain.

    \vspace{1.5em}
    The Transfer Function is found via voltage division with impedances:
    \[
      H(s) \;=\; \frac{V_C(s)}{V_s(s)}
             \;=\;
             \frac{R_2 \,\|\, \frac{1}{sC}}
                  {R_1 + \left(R_2 \,\|\, \frac{1}{sC}\right)}
    \]

    \vspace{1.5em}
    Simplifying and substituting values yields:
    \[
      H(s) \;=\; \frac{12.5}{s + \mathbf{25}}
    \]

    \vspace{1.5em}
    \normalsize
    \textit{The pole is located at $s = -25\,\mathrm{rad/s}$, which defines the
    system's cutoff frequency.}
    \vfill
  \end{center}
\end{frame}

%% ── ADVANCED: Laplace Transform ──────────────────────────────────
\begin{frame}{\textbf{Advanced: Laplace Transform}}
  \begin{center}
    \vfill
    In the \textbf{$s$-domain}, the input is a step: $V_s(s) = \dfrac{10}{s}$.

    \vspace{1.5em}
    Using Nodal Analysis and solving for $V_C(s)$:
    \[
      V_C(s) \;=\; \frac{125}{s(s + 25)}
    \]

    \vspace{1.5em}
    Applying \textbf{Partial Fraction Expansion}:
    \[
      V_C(s) \;=\; \frac{5}{s} \;-\; \frac{5}{s + 25}
    \]

    \vspace{2em}
    \Large\textbf{\textcolor{madblue}{$V_C(t) \;=\; 5\bigl(1 - e^{-25t}\bigr)\,u(t)\;\mathrm{V}$}}
    \vfill
  \end{center}
\end{frame}

%% ── ADVANCED: State-Space ────────────────────────────────────────
\begin{frame}{\textbf{Advanced: State-Space Formulation}}
  \begin{center}
    \vfill
    For modern control theory, we express the circuit as:
    \[
      \mathbf{\dot{x}(t) \;=\; Ax(t) + Bu(t)}
    \]

    \vspace{1.5em}
    Let the state variable be $x(t) = V_C(t)$ and the input $u(t) = V_s(t)$.

    \vspace{1.5em}
    From the earlier ODE, we directly extract the matrices:
    \[
      \mathbf{\dot{x}(t) \;=\;
        \underbrace{-25}_{\mathbf{A}} \, x(t) \;+\;
        \underbrace{12.5}_{\mathbf{B}} \, u(t)}
    \]

    \vspace{1.5em}
    \normalsize
    \textit{The eigenvalue of $\mathbf{A}$ is $\lambda = -25$.\\
    Since $\lambda < 0$, the system is \textbf{strictly stable}.}
    \vfill
  \end{center}
\end{frame}

%% ── ADVANCED: Energy Dynamics ────────────────────────────────────
\begin{frame}{\textbf{Advanced: Energy Dynamics}}
  \begin{center}
    \vfill
    \textbf{Steady-State Energy Stored ($t \to \infty$):}
    \[
      E_C \;=\; \frac{1}{2} C V_\infty^2
           \;=\; \frac{1}{2} (4\,\mu\mathrm{F})(5\,\mathrm{V})^2
           \;=\; \mathbf{50\,\mu\mathrm{J}}
    \]

    \vspace{2em}
    \textbf{Instantaneous Power Delivered:}
    \[
      p_C(t) \;=\; V_C(t) \cdot i_C(t)
    \]

    \vspace{1em}
    \[
      p_C(t) \;=\; 2.5\bigl(e^{-25t} - e^{-50t}\bigr)\;\mathrm{mW}
    \]

    \vspace{1.5em}
    \normalsize
    \textit{Maximum power transfer occurs at
    $t = \dfrac{\ln 2}{25} \approx \mathbf{27.7\,\mathrm{ms}}$.}
    \vfill
  \end{center}
\end{frame}

%% ── RK4: Theory Part 1 ───────────────────────────────────────────
\begin{frame}{\textbf{Numerical Verification: RK4 Method (1/3)}}
  \begin{center}
    \vfill
    \textbf{Why Runge-Kutta 4 (RK4)?}\\
    Analytical solutions are not always feasible for complex circuits. Numerical methods like RK4 provide highly accurate approximations with a global error of $\mathcal{O}(h^4)$.

    \vspace{1.5em}
    The governing \textbf{Ordinary Differential Equation (ODE)} for our circuit is:
    \[
      \frac{dV_C}{dt} \;=\; f(t,\,V_C) \;=\; 125 - 25\,V_C
    \]

    \vspace{1em}
    We evaluate the voltage $V_C$ at discrete time steps $t_n$, such that:
    \[
      t_{n+1} = t_n + h
    \]
    where $h$ is the chosen \textbf{time step}.
    \vfill
  \end{center}
\end{frame}

%% ── RK4: Theory Part 2 ───────────────────────────────────────────
\begin{frame}{\textbf{Numerical Verification: RK4 Method (2/3)}}
  \begin{center}
    \vfill
    RK4 calculates four different \textbf{slopes} ($k_1, k_2, k_3, k_4$) across the interval $[t_n, t_{n+1}]$ to find a highly accurate weighted average.

    \vspace{1em}
    \begin{align*}
      k_1 &= f(t_n,\;V_n) \\
      &\small\textit{(Slope at the beginning of the interval)} \\[0.8em]
      k_2 &= f\!\left(t_n+\tfrac{h}{2},\;V_n+\tfrac{h}{2}k_1\right) \\
      &\small\textit{(Slope at the midpoint, estimated using $k_1$)} \\[0.8em]
      k_3 &= f\!\left(t_n+\tfrac{h}{2},\;V_n+\tfrac{h}{2}k_2\right) \\
      &\small\textit{(Slope at the midpoint, refined using $k_2$)} \\[0.8em]
      k_4 &= f(t_n+h,\;V_n+h\,k_3) \\
      &\small\textit{(Slope at the end of the interval, estimated using $k_3$)}
    \end{align*}
    \vfill
  \end{center}
\end{frame}

%% ── RK4: Theory Part 3 ───────────────────────────────────────────
\begin{frame}{\textbf{Numerical Verification: RK4 Method (3/3)}}
  \begin{center}
    \vfill
    \textbf{The Final Update Rule:}
    The next voltage value $V_{n+1}$ is computed by adding a weighted average of the four slopes to the current value $V_n$:
    \[
      V_{n+1} \;=\; V_n \;+\; \frac{h}{6}\bigl(k_1 + 2k_2 + 2k_3 + k_4\bigr)
    \]

    \vspace{1.5em}
    \textbf{Implementation Details for our Circuit:}
    \begin{itemize}
        \item \textbf{Initial Condition:} $t_0 = 0$, $V_C(0) = 0\,\mathrm{V}$
        \item \textbf{Step Size:} $h = 0.5 \times 10^{-3}\,\mathrm{s}$ ($0.5\,\mathrm{ms}$)
        \item \textbf{Simulation Time:} $t_{\text{end}} = 200\,\mathrm{ms}$
    \end{itemize}

    \vspace{1em}
    \normalsize\textit{Greater weight is given to the midpoint slopes ($k_2, k_3$) as they better represent the function's average behavior over the given interval.}
    \vfill
  \end{center}
\end{frame}

%% ── RK4: Python Code ─────────────────────────────────────────────
\begin{frame}[fragile,allowframebreaks]{\textbf{RK4 Verification -- Python Code}}
\begin{lstlisting}
import numpy as np
import matplotlib.pyplot as plt

# ── Circuit parameters ──────────────────────────────────────────
Vs  = 10.0       # Source voltage  (V)
R1  = 20e3       # Series resistor (Ohm)
R2  = 20e3       # Shunt  resistor (Ohm)
C   = 4e-6       # Capacitance     (F)
Rth = R1 * R2 / (R1 + R2)          # Thevenin resistance
Vinf = Vs * R2 / (R1 + R2)         # Steady-state voltage

# ODE:  dVc/dt = f(Vc)  =  125 - 25*Vc
def f(Vc):
    return (Vinf - Vc) / (Rth * C)  # = 125 - 25*Vc

# ── RK4 solver ──────────────────────────────────────────────────
h    = 0.5e-3      # step size (0.5 ms)
t_end = 0.200      # end time  (200 ms)
N    = int(t_end / h)

t_rk = np.zeros(N + 1)
Vc_rk = np.zeros(N + 1)
Vc_rk[0] = 0.0    # initial condition

for n in range(N):
    k1 = f(Vc_rk[n])
    k2 = f(Vc_rk[n] + h/2 * k1)
    k3 = f(Vc_rk[n] + h/2 * k2)
    k4 = f(Vc_rk[n] + h     * k3)
    Vc_rk[n+1] = Vc_rk[n] + h/6 * (k1 + 2*k2 + 2*k3 + k4)
    t_rk[n+1]  = t_rk[n] + h

# ── Exact (theoretical) solution ────────────────────────────────
t_ex  = np.linspace(0, t_end, 2000)
Vc_ex = Vinf * (1 - np.exp(-25 * t_ex))
ic_ex = (Vinf / Rth) * np.exp(-25 * t_ex) * 1000  # mA
ic_rk = np.array([f(v) * C * 1000 for v in Vc_rk]) # mA

# ── Plot: Vc comparison ─────────────────────────────────────────
fig, axes = plt.subplots(1, 2, figsize=(12, 4))

ax = axes[0]
ax.plot(t_ex*1000,  Vc_ex,  'b-',  lw=2,   label='Exact: $5(1-e^{-25t})$')
ax.plot(t_rk*1000,  Vc_rk,  'r--', lw=1.5, label='RK4 (h=0.5 ms)')
ax.axhline(5, color='gray', ls=':', lw=1)
ax.set_xlabel('Time (ms)')
ax.set_ylabel('$V_C$ (V)')
ax.set_title('Capacitor Voltage')
ax.legend()
ax.grid(True, alpha=0.3)

# ── Plot: ic comparison ─────────────────────────────────────────
ax2 = axes[1]
ax2.plot(t_ex*1000,  ic_ex,  'b-',  lw=2,   label='Exact: $0.5\,e^{-25t}$ mA')
ax2.plot(t_rk*1000,  ic_rk,  'r--', lw=1.5, label='RK4 (h=0.5 ms)')
ax2.axhline(0, color='gray', ls=':', lw=1)
ax2.set_xlabel('Time (ms)')
ax2.set_ylabel('$i_c$ (mA)')
ax2.set_title('Capacitor Current')
ax2.legend()
ax2.grid(True, alpha=0.3)

plt.suptitle('RK4 vs Exact Solution', fontsize=13, fontweight='bold')
plt.tight_layout()
plt.savefig('figs/rk4.png', dpi=150)
plt.show()

# ── Max absolute error ──────────────────────────────────────────
Vc_ex_at_rk = Vinf * (1 - np.exp(-25 * t_rk))
err = np.max(np.abs(Vc_rk - Vc_ex_at_rk))
print(f"Max |error| in Vc: {err*1000:.4f} mV")
\end{lstlisting}
\end{frame}

t = np.linspace(0, 0.2, 1000)

Vc     = V * (1 - np.exp(-t/tau))
dVc_dt = (V / tau) * np.exp(-t/tau)

bg     = '#0b1220'
curve  = '#8fd3d1'
accent = '#f5e6a8'

fig, ax = plt.subplots(figsize=(7, 5))
fig.patch.set_facecolor(bg)
ax.set_facecolor(bg)
ax.plot(Vc, dVc_dt, color=curve, lw=2)
ax.fill_between(Vc, dVc_dt, alpha=0.12, color=curve)
ax.scatter([0], [125], color=accent, zorder=5, s=60,
           label='Initial ($V_C=0$, $dV/dt=125$)')
ax.scatter([5], [0],   color=accent, zorder=5, s=60,
           label='Equilibrium ($V_C=5$ V)')
ax.set_xlabel('Capacitor Voltage $V_C$ (V)', color='white')
ax.set_ylabel('Rate of Change $dV_C/dt$ (V/s)', color='white')
ax.set_title('Phase Portrait: $dV_C/dt$ vs $V_C$', color='white')
ax.tick_params(colors='white')
ax.legend(facecolor=bg, labelcolor='white')
ax.grid(True, alpha=0.2)
plt.savefig('figs/PP.png', dpi=150, facecolor=bg)
plt.show()
