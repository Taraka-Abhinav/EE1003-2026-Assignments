\begin{frame}{\textbf{Problem 1.2.16 \;--\; EC 2007}}

  \textbf{\large The equation}
  \[
    \bm{f(x) = x^2 + x^4 - 4x - 4 = 0}
  \]
  \textbf{is to be solved using Newton-Raphson.
  If $x = 2$ is the initial approximation, then the
  \emph{next} approximation is:}

  \vspace{0.8em}
  \begin{columns}[c]
    \begin{column}{0.48\textwidth}
      \myopt{a}{$\dfrac{2}{3}$}
      \vspace{0.5em}
      \myopt{b}{$\dfrac{4}{3}$}
    \end{column}
    \begin{column}{0.48\textwidth}
      \myopt{c}{$1$}
      \vspace{0.5em}
      \myopt{d}{$\dfrac{3}{2}$}
    \end{column}
  \end{columns}

\end{frame}

%% ── NR FORMULA ────────────────────────────────────────────
\begin{frame}{\textbf{The Newton-Raphson Formula}}
  \begin{center}
    \vfill

    The Newton-Raphson iteration formula is:
    \[
      x_{n+1} \;=\; x_n \;-\; \frac{f(x_n)}{f'(x_n)}
    \]

    \vspace{1em}
    For \textbf{one} iteration from $x_0 = 2$:
    \[
      x_1 \;=\; x_0 \;-\; \frac{f(x_0)}{f'(x_0)}
            \;=\; 2 \;-\; \frac{f(2)}{f'(2)}
    \]

    \vspace{1.2em}
    \textit{We need two things:} \quad
    \textcolor{madblue}{\textbf{$f(2)$}} \quad and \quad
    \textcolor{madblue}{\textbf{$f'(2)$}}

    \vfill
  \end{center}
\end{frame}

%% ── DERIVATIVE ────────────────────────────────────────────
\begin{frame}{\textbf{Setting Up \;--\; Function \& Derivative}}
  \begin{center}
    \vfill

    The function:
    \[
      f(x) \;=\; x^4 + x^2 - 4x - 4
    \]

    \vspace{0.8em}
    Differentiating term by term:
    \[
      f'(x) \;=\; 4x^3 + 2x - 4
    \]

    \vspace{1.2em}
    \renewcommand{\arraystretch}{1.3}
    \begin{tabular}{r@{\qquad}l}
      \textbf{Term} & \textbf{Derivative} \\
      $x^4$         & $4x^3$ \\
      $x^2$         & $2x$   \\
      $-4x$         & $-4$   \\
      $-4$          & $0$    \\
    \end{tabular}

    \vfill
  \end{center}
\end{frame}

    \[
      f(2) \;=\; (2)^4 + (2)^2 - 4(2) - 4
    \]
    \[

    \]

    \LARGE\textbf{\textcolor{madblue}{$f(2) \;=\; 8$}}

    \vfill
  \end{center}
\end{frame}

  \begin{center}

      f'(x) = 4x^3 + 2x - 4
    \]

  \end{center}
\end{frame}

%% ── ADVANCED B.TECH/M.TECH THEORY DETAILS ─────────────────

\begin{frame}{\textbf{Derivation via Taylor Series}}
  \begin{center}
  \vfill
  Let $x^*$ be the exact root, meaning $f(x^*) = 0$. \\
  Expanding $f(x)$ using the Taylor series around our guess $x_n$:
  \[
    f(x^*) = f(x_n) + (x^* - x_n)f'(x_n) + \frac{(x^* - x_n)^2}{2!}f''(\xi) = 0
  \]
  For a small interval, we truncate the higher-order terms ($x^* \approx x_{n+1}$):
  \[
    f(x_n) + (x_{n+1} - x_n)f'(x_n) \approx 0
  \]
  \vspace{0.5em}
  Rearranging algebraically to solve for the next iteration $x_{n+1}$:
  \[
    \boxed{ x_{n+1} = x_n - \frac{f(x_n)}{f'(x_n)} }
  \]
  \vfill
  \end{center}
\end{frame}

\begin{frame}{\textbf{Error Analysis \& Asymptotic Constant}}
  \begin{center}
  \vfill
  Let the error at step $n$ be $e_n = x_n - x^*$. \\
  Using the exact Taylor expansion, we find the relation:
  \[
    e_{n+1} \approx \frac{f''(x^*)}{2 f'(x^*)} e_n^2 
  \]
  \vspace{0.5em}
  For a simple root ($f'(x^*) \neq 0$), the sequence exhibits \textbf{Quadratic Convergence} ($p=2$). The \textbf{Asymptotic Error Constant ($M$)} is defined as:
  \[
    \lim_{n \to \infty} \frac{|e_{n+1}|}{|e_n|^2} = M = \left| \frac{f''(x^*)}{2f'(x^*)} \right|
  \]
  \vspace{0.5em}
  \textit{Practical Meaning:} The number of correct decimal places essentially \textbf{doubles} with every single iteration once $x_n$ is close to $x^*$.
  \vfill
  \end{center}
\end{frame}

\begin{frame}{\textbf{Conditions for Guaranteed Convergence}}
  \begin{center}
  \vfill
  \textbf{Fourier's Sufficient Condition:} Convergence is mathematically guaranteed if the initial guess $x_0$ satisfies:
  \[
    \left| \frac{f(x) f''(x)}{[f'(x)]^2} \right| < 1 \quad \text{for all } x \text{ in the interval.}
  \]
  \vspace{0.2em}
  \textbf{Verifying our problem at $x_0 = 2$:}
  \[ f(2) = 8, \quad f'(2) = 32 \]
  \[ f''(x) = 12x^2 + 2 \implies f''(2) = 12(2)^2 + 2 = 50 \]
  \[
    \left| \frac{f(2) f''(2)}{[f'(2)]^2} \right| = \frac{8 \times 50}{(32)^2} = \frac{400}{1024} \approx 0.39 < 1
  \]
  \textbf{\textcolor{okgreen}{Condition strongly satisfied!}} Convergence from $x_0=2$ is absolute.
  \vfill
  \end{center}
\end{frame}

\begin{frame}{\textbf{Advanced Extensions: Multidimensional NR}}
  \begin{center}
  \vfill
  In M.Tech/Advanced applications (e.g., Load Flow Analysis in Power Systems), we solve systems of non-linear equations $\mathbf{F}(\mathbf{x}) = 0$.
  \vspace{1em}
  
  The scalar derivative $f'(x)$ is replaced by the \textbf{Jacobian Matrix} $\mathbf{J}(\mathbf{x})$, containing partial derivatives:
  \[
    \mathbf{J}(\mathbf{x}) = \begin{bmatrix}
    \frac{\partial f_1}{\partial x_1} & \cdots & \frac{\partial f_1}{\partial x_n} \\
    \vdots & \ddots & \vdots \\
    \frac{\partial f_n}{\partial x_1} & \cdots & \frac{\partial f_n}{\partial x_n}
    \end{bmatrix}
  \]
  \vspace{1em}
  The iteration formula becomes a matrix operation:
  \[
    \mathbf{x}_{n+1} = \mathbf{x}_n - \mathbf{J}^{-1}(\mathbf{x}_n) \mathbf{F}(\mathbf{x}_n)
  \]
  \vfill
  \end{center}
\end{frame}

\begin{frame}{\textbf{Handling Pathologies: Multiple Roots}}
  \begin{center}
  \vfill
  If a root has a multiplicity $m > 1$ (i.e., $f(x^*) = 0$ and $f'(x^*) = 0$), standard Newton-Raphson degrades to \textbf{Linear Convergence}.
  \vspace{1em}
  
  To restore quadratic convergence, we use the \textbf{Modified Newton-Raphson Method}:
  \[
    x_{n+1} = x_n - m \frac{f(x_n)}{f'(x_n)}
  \]
  \vspace{1em}
  Alternatively, applying standard NR to the function $u(x) = \frac{f(x)}{f'(x)}$ always yields quadratic convergence, regardless of root multiplicity.
  \vfill
  \end{center}
\end{frame}

\begin{frame}{\textbf{Global Dynamics: Basins of Attraction}}
  \begin{center}
  \vfill
  Newton's method is highly sensitive to the initial guess $x_0$. 
  \vspace{1em}
  
  When extended to the \textbf{Complex Plane} ($z \in \mathbb{C}$), the roots act as "attractors". The set of all initial guesses that converge to a specific root is called its \textbf{Basin of Attraction}.
  \vspace{1em}
  
  For our function $f(z) = z^4 + z^2 - 4z - 4$, coloring the complex plane based on which of the 4 roots an initial guess converges to generates a \textbf{Newton Fractal}. The boundaries between these basins are infinitely complex and chaotic.
  \vfill
  \end{center}
\end{frame}

import numpy as np
import matplotlib.pyplot as plt

def f(x):
    return x**4 + x**2 - 4*x - 4

def df(x):
    return 4*x**3 - 2*x - 4

x0 = 2
x1 = x0 - f(x0)/df(x0)

plt.style.use('dark_background')

fig1, ax1 = plt.subplots(figsize=(8,5))

x = np.linspace(0, 3, 400)
ax1.plot(x, f(x))

tangent = f(x0) + df(x0)*(x - x0)
ax1.plot(x, tangent, '--')
\end{lstlisting}

import numpy as np
import matplotlib.pyplot as plt

def f(x):
    return x**4 + x**2 - 4*x - 4

def df(x):
    return 4*x**3 + 2*x - 4

true_root = 1.676251739196307

x = 2.0
errors = []
iterations = []

for i in range(7):
    errors.append(abs(x - true_root))
    iterations.append(i)
    x = x - f(x)/df(x)
\end{lstlisting}
\end{frame}
